let transactions = []; const products = [ { name: "S1 PRO HODL", price: 50, daily: 7, total: 315 }, { name: "S1K PRO HODL", price: 150, daily: 24, total: 1080 }, { name: "S2 PRO HODL", price: 350, daily: 63, total: 2835 }, { name: "S2K PRO HODL", price: 700, daily: 140, total: 6300 }, { name: "S3 PRO HODL", price: 1500, daily: 310, total: 13950 } ]; function login() { window.location.href = "dashboard.html"; } function register() { alert("Registration is simulated. Please log in."); } function recharge() { const amt = parseFloat(document.getElementById("amount").value); if (!amt || amt <= 0) return alert("Enter a valid amount"); balance += amt; updateBalance(); addTransaction(`Recharged GHS ${amt.toFixed(2)}`); document.getElementById("amount").value = ""; } function updateBalance() { const el = document.getElementById("balance"); if (el) el.textContent = `GHS ${balance.toFixed(2)}`; } function addTransaction(text) { transactions.unshift(text); const list = document.getElementById("transactions"); if (list) { list.innerHTML = transactions.map(t => `
  • ${t}
    • `).join(""); } } function renderProducts() { const table = document.getElementById("productTable"); if (!table) return; table.innerHTML = products.map((p, i) => ` ${p.name} GHS ${p.price} GHS ${p.daily} GHS ${p.total} `).join(""); } function buyProduct(index) { const product = products[index]; if (balance < product.price) return alert("Not enough balance!"); balance -= product.price; updateBalance(); addTransaction(`Purchased ${product.name} for GHS ${product.price}`); } // Auto-render if on dashboard window.onload = () => { updateBalance(); renderProducts(); }; ``` --- ✅ **How to use:** 1.
